    <script>
        // 1. 生成随机爱心
        function createHeart() {
            const heart = document.createElement('div');
            heart.className = 'heart';
            
            // 随机位置、大小、动画时长
            const left = Math.random() * 100; // 水平位置 0-100%
            const size = Math.random() * 15 + 10; // 大小 10-25px
            const duration = Math.random() * 6 + 4; // 动画时长 4-10秒
            
            // 设置样式
            heart.style.left = `${left}%`;
            heart.style.width = `${size}px`;
            heart.style.height = `${size}px`;
            heart.style.animationDuration = `${duration}s`;
            heart.style.animationDelay = `${Math.random() * 5}s`;
            
            // 添加到页面
            const container = document.getElementById('heartContainer');
            container.appendChild(heart);
            
            // 移除过期的爱心
            setTimeout(() => {
                heart.remove();
            }, duration * 1000);
        }

        // 2. 定时生成爱心
        setInterval(createHeart, 300);

        // 3. 按钮点击事件
        const loveBtn = document.getElementById('loveBtn');
        loveBtn.addEventListener('click', function() {
            // 播放音乐（需先替换音乐链接）
            const music = document.getElementById('music');
            music.play().catch(err => console.log('音乐播放需要用户交互:', err));
            
            // 修改文字
            document.querySelector('.title').innerText = '💖 双向奔赴的爱 💖';
            document.querySelector('.content').innerHTML = 
                '余生漫漫，皆是你<br>' +
                '愿我们岁岁年年，朝朝暮暮<br>' +
                '爱你不止一点点 ❤️';
            loveBtn.innerText = '永远在一起 💞';
            loveBtn.disabled = true;
            loveBtn.style.backgroundColor = '#e62958';
        });

        // 4. 页面加载时自动播放音乐（微信中需用户点击后才能播放）
        document.addEventListener('touchstart', function() {
            const music = document.getElementById('music');
            music.play().catch(err => {});
        }, { once: true });
    </script>
